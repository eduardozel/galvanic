import 'package:wifi_iot/wifi_iot.dart';  // –î–ª—è WiFi

void main() {
  runApp(const MyApp());
}

class MyApp extends StatelessWidget {
  const MyApp({super.key});

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: 'Vase Controller',
      theme: ThemeData(
        primarySwatch: Colors.blue,
      ),
      home: const HomePage(),
    );
  }
}

class HomePage extends StatefulWidget {
  const HomePage({super.key});

  @override
  State<HomePage> createState() => _HomePageState();
}

class _HomePageState extends State<HomePage> {
  IOWebSocketChannel? _channel;
  String _status = '–ü–æ–¥–∫–ª—é—á–µ–Ω–∏–µ –∫ WiFi...';
  int _remainingMinutes = 0;  // –î–ª—è –æ—Ç–æ–±—Ä–∞–∂–µ–Ω–∏—è –æ—Å—Ç–∞–≤—à–µ–≥–æ—Å—è –≤—Ä–µ–º–µ–Ω–∏ –≤ –º–∏–Ω—É—Ç–∞—Ö
  Timer? _reconnectTimer;

  @override
  void initState() {
    super.initState();
    _connectToWifiAndWebSocket();
  }

  @override
  void dispose() {
    _channel?.sink.close();
    _reconnectTimer?.cancel();
    super.dispose();
  }

  // –ü–æ–¥–∫–ª—é—á–µ–Ω–∏–µ –∫ WiFi –∏ WebSocket
  Future<void> _connectToWifiAndWebSocket() async {
    try {
      // –ü—Ä–æ–≤–µ—Ä—è–µ–º –∏ –ø–æ–¥–∫–ª—é—á–∞–µ–º—Å—è –∫ WiFi
      bool isConnected = await WiFiForIoTPlugin.isConnected();
      if (!isConnected) {
        String ssid = 'vase';
        String password = 'vase23456';
        bool connected = await WiFiForIoTPlugin.connect(
          ssid,
          password: password,
          security: NetworkSecurity.WPA,
        );
        if (!connected) {
          setState(() => _status = '–û—à–∏–±–∫–∞ –ø–æ–¥–∫–ª—é—á–µ–Ω–∏—è –∫ WiFi');
          return;
        }
      }
      setState(() => _status = '–ü–æ–¥–∫–ª—é—á–µ–Ω–æ –∫ WiFi. –ü–æ–¥–∫–ª—é—á–µ–Ω–∏–µ –∫ WebSocket...');

      // –ü–æ–¥–∫–ª—é—á–µ–Ω–∏–µ –∫ WebSocket
      _channel = IOWebSocketChannel.connect('ws://192.168.4.1/ws');
      _channel!.stream.listen(
        (message) {
          _handleIncomingMessage(message);  // –û–±—Ä–∞–±–æ—Ç–∫–∞ –≤—Ö–æ–¥—è—â–µ–≥–æ —Å–æ–æ–±—â–µ–Ω–∏—è
        },
        onError: (error) {
          setState(() => _status = '–û—à–∏–±–∫–∞ WebSocket: $error');
          _reconnect();
        },
        onDone: () {
          setState(() => _status = 'WebSocket –∑–∞–∫—Ä—ã—Ç');
          _reconnect();
        },
      );

      // –û—Ç–ø—Ä–∞–≤–ª—è–µ–º –∑–∞–ø—Ä–æ—Å –Ω–∞ —Ç–µ–∫—É—â–µ–µ —Å–æ—Å—Ç–æ—è–Ω–∏–µ –ø–æ—Å–ª–µ –ø–æ–¥–∫–ª—é—á–µ–Ω–∏—è
      _channel!.sink.add(jsonEncode({'act': 'getState'}));
      setState(() => _status = '–ü–æ–¥–∫–ª—é—á–µ–Ω–æ –∫ WebSocket');
    } catch (e) {
      setState(() => _status = '–û—à–∏–±–∫–∞: $e');
    }
  }

  // –û–±—Ä–∞–±–æ—Ç–∫–∞ –≤—Ö–æ–¥—è—â–µ–≥–æ —Å–æ–æ–±—â–µ–Ω–∏—è: –ø–∞—Ä—Å–∏–Ω–≥ JSON –∏ –æ–±–Ω–æ–≤–ª–µ–Ω–∏–µ —Ç–∞–π–º–µ—Ä–∞
  void _handleIncomingMessage(dynamic message) {
    try {
      final data = jsonDecode(message as String);
      if (data.containsKey('timer')) {
        int timerSeconds = data['timer'] as int;
        setState(() {
          _remainingMinutes = (timerSeconds / 60).floor();  // –ö–æ–Ω–≤–µ—Ä—Ç–∞—Ü–∏—è –≤ –º–∏–Ω—É—Ç—ã
          _status = '–ü–æ–ª—É—á–µ–Ω–æ –æ–±–Ω–æ–≤–ª–µ–Ω–∏–µ';
        });
      }
    } catch (e) {
      setState(() => _status = '–û—à–∏–±–∫–∞ –ø–∞—Ä—Å–∏–Ω–≥–∞: $e');
    }
  }

  // –ü–µ—Ä–µ–ø–æ–¥–∫–ª—é—á–µ–Ω–∏–µ WebSocket —Å –∑–∞–¥–µ—Ä–∂–∫–æ–π
  void _reconnect() {
    _reconnectTimer?.cancel();
    _reconnectTimer = Timer(const Duration(seconds: 5), _connectToWifiAndWebSocket);
  }

  // –û—Ç–ø—Ä–∞–≤–∫–∞ –∫–æ–º–∞–Ω–¥—ã "–í–∫–ª—é—á–∏—Ç—å" (start, white, duration 300)
  void _sendOnCommand() {
    if (_channel != null) {
      final command = jsonEncode({
        'act': 'start',
        'mode': 'white',
        'duration': 300,  // 300 —Å–µ–∫—É–Ω–¥
      });
      _channel!.sink.add(command);
      setState(() => _status = '–ö–æ–º–∞–Ω–¥–∞ "–í–∫–ª—é—á–∏—Ç—å" –æ—Ç–ø—Ä–∞–≤–ª–µ–Ω–∞');
    } else {
      setState(() => _status = '–ù–µ –ø–æ–¥–∫–ª—é—á–µ–Ω–æ');
    }
  }

  // –û—Ç–ø—Ä–∞–≤–∫–∞ –∫–æ–º–∞–Ω–¥—ã "–í—ã–∫–ª—é—á–∏—Ç—å" (stop)
  void _sendOffCommand() {
    if (_channel != null) {
      final command = jsonEncode({
        'act': 'stop',
      });
      _channel!.sink.add(command);
      setState(() => _status = '–ö–æ–º–∞–Ω–¥–∞ "–í—ã–∫–ª—é—á–∏—Ç—å" –æ—Ç–ø—Ä–∞–≤–ª–µ–Ω–∞');
    } else {
      setState(() => _status = '–ù–µ –ø–æ–¥–∫–ª—é—á–µ–Ω–æ');
    }
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: const Text('Vase Controller'),
      ),
      body: Center(
        child: Column(
          mainAxisAlignment: MainAxisAlignment.center,
          children: [
            // –°—Ç–∞—Ç—É—Å
            Text(_status, style: const TextStyle(fontSize: 18)),
            const SizedBox(height: 20),
            // –û—Ç–æ–±—Ä–∞–∂–µ–Ω–∏–µ –æ—Å—Ç–∞–≤—à–µ–≥–æ—Å—è –≤—Ä–µ–º–µ–Ω–∏
            Text(
              '–û—Å—Ç–∞–≤—à–µ–µ—Å—è –≤—Ä–µ–º—è: $_remainingMinutes –º–∏–Ω',
              style: const TextStyle(fontSize: 24, fontWeight: FontWeight.bold),
            ),
            const SizedBox(height: 40),
            // –ö–Ω–æ–ø–∫–∞ "–í–∫–ª—é—á–∏—Ç—å" (–∑–µ–ª—ë–Ω–∞—è –∏–∫–æ–Ω–∫–∞ –ª–∞–º–ø–æ—á–∫–∏)
            IconButton(
              icon: const Icon(Icons.lightbulb, color: Colors.green, size: 100),
              onPressed: _sendOnCommand,
            ),
            const SizedBox(height: 20),
            // –ö–Ω–æ–ø–∫–∞ "–í—ã–∫–ª—é—á–∏—Ç—å" (–∫—Ä–∞—Å–Ω–∞—è –∏–∫–æ–Ω–∫–∞ –ª–∞–º–ø–æ—á–∫–∏)
            IconButton(
              icon: const Icon(Icons.lightbulb_outline, color: Colors.red, size: 100),
              onPressed: _sendOffCommand,
            ),
          ],
        ),
      ),
    );
  }
}
