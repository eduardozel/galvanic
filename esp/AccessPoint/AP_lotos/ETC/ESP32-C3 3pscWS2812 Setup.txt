   #include "esp_log.h"

   #define EXAMPLE_LED_NUMBERS         9       // –í—Å–µ–≥–æ 9 LED (3 –¥–∏—Å–∫–∞ –ø–æ 3)
   #define EXAMPLE_CHASE_SPEED_MS      100     // –°–∫–æ—Ä–æ—Å—Ç—å –∞–Ω–∏–º–∞—Ü–∏–∏

       rmt_channel_handle_t led_chan = NULL;

           .gpio_num = 4,                  // GPIO –ø–∏–Ω (–∏–∑–º–µ–Ω–∏—Ç–µ, –µ—Å–ª–∏ –¥—Ä—É–≥–æ–π)

                   led_strip_pixels[led * 3 + 2] = 0;   // B
               } else if (base == 1) { // –í—Ç–æ—Ä–æ–π: –∑–µ–ª—ë–Ω—ã–π
                   led_strip_pixels[led * 3 + 0] = 50;  // G
                   led_strip_pixels[led * 3 + 1] = 0;   // R
                   led_strip_pixels[led * 3 + 2] = 0;   // B
               } else { // –¢—Ä–µ—Ç–∏–π: —Å–∏–Ω–∏–π
                   led_strip_pixels[led * 3 + 0] = 0;   // G
                   led_strip_pixels[led * 3 + 1] = 0;   // R
                   led_strip_pixels[led * 3 + 2] = 50;  // B
               }
