  - `remain_time`: –û—Å—Ç–∞–≤—à–µ–µ—Å—è –≤—Ä–µ–º—è —Å–≤–µ—á–µ–Ω–∏—è. –ò–∑–Ω–∞—á–∞–ª—å–Ω–æ —Ä–∞–≤–Ω–æ `max_time`. –û–±–Ω–æ–≤–ª—è–µ—Ç—Å—è –ø—Ä–∏ –≤—ã–∫–ª—é—á–µ–Ω–∏–∏ LED (–ø–æ –∫–Ω–æ–ø–∫–µ –∏–ª–∏ —Ç–∞–π–º–∞—É—Ç—É) –∏ –∑–∞–ø–∏—Å—ã–≤–∞–µ—Ç—Å—è –≤ —Ñ–∞–π–ª.
- **–ü—Ä–∏ —Å—Ç–∞—Ä—Ç–µ**:
  - –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∏—Ä—É–µ–º SPIFFS.
  - –°—á–∏—Ç—ã–≤–∞–µ–º `max_time` –∏ `remain_time` –∏–∑ —Ñ–∞–π–ª–∞.
- **–ü–æ –∫–∞—Å–∞–Ω–∏—é –∫–Ω–æ–ø–∫–∏** (–≤ –ø—Ä–µ—Ä—ã–≤–∞–Ω–∏–∏ —Å debounce):
  - –ï—Å–ª–∏ LED –≤—ã–∫–ª—é—á–µ–Ω: –í–∫–ª—é—á–∞–µ–º LED –∏ –∑–∞–ø—É—Å–∫–∞–µ–º —Ç–∞–π–º–µ—Ä –Ω–∞ `remain_time` —Å–µ–∫—É–Ω–¥. –ù–∞—á–∏–Ω–∞–µ–º –æ—Ç—Å—á—ë—Ç (–∑–∞–ø–æ–º–∏–Ω–∞–µ–º start_time).
  - –ï—Å–ª–∏ LED –≤–∫–ª—é—á–µ–Ω: –í—ã–∫–ª—é—á–∞–µ–º LED, —Ä–∞—Å—Å—á–∏—Ç—ã–≤–∞–µ–º elapsed_time (—Å–∫–æ–ª—å–∫–æ —Å–≤–µ—Ç–∏–ª), –æ–±–Ω–æ–≤–ª—è–µ–º `remain_time -= elapsed_time`, –∑–∞–ø–∏—Å—ã–≤–∞–µ–º `remain_time` –≤ —Ñ–∞–π–ª.
- **–¢–∞–π–º–µ—Ä —Å–≤–µ—á–µ–Ω–∏—è**:

  - –ö–æ–≥–¥–∞ —Ç–∞–π–º–µ—Ä —Å—Ä–∞–±–∞—Ç—ã–≤–∞–µ—Ç: –í—ã–∫–ª—é—á–∞–µ–º LED, —É—Å—Ç–∞–Ω–∞–≤–ª–∏–≤–∞–µ–º `remain_time = 0`, –∑–∞–ø–∏—Å—ã–≤–∞–µ–º –≤ —Ñ–∞–π–ª.
- **–û–≥—Ä–∞–Ω–∏—á–µ–Ω–∏—è**:
  - LED –Ω–µ —Å–≤–µ—Ç–∏—Ç –¥–æ–ª—å—à–µ `remain_time` (–∫–æ—Ç–æ—Ä–æ–µ <= `max_time`).
  - –ü—Ä–∏ —Å–ª–µ–¥—É—é—â–µ–º –∫–∞—Å–∞–Ω–∏–∏: –ï—Å–ª–∏ `remain_time > 0`, –≤–∫–ª—é—á–∞–µ–º –Ω–∞ –æ—Å—Ç–∞–≤—à–µ–µ—Å—è –≤—Ä–µ–º—è. –ï—Å–ª–∏ 0, –Ω–∏—á–µ–≥–æ –Ω–µ –ø—Ä–æ–∏—Å—Ö–æ–¥–∏—Ç (–∏–ª–∏ –º–æ–∂–Ω–æ –¥–æ–±–∞–≤–∏—Ç—å –ª–æ–≥–∏–∫—É —Å–±—Ä–æ—Å–∞, –Ω–æ –ø–æ –∑–∞–ø—Ä–æ—Å—É ‚Äî –ø—Ä–æ—Å—Ç–æ –Ω–µ –≤–∫–ª—é—á–∞–µ–º).
  - "–î–ª–∏—Ç–µ–ª—å–Ω–æ—Å—Ç—å —Å–≤–µ—á–µ–Ω–∏—è" –∏–Ω—Ç–µ—Ä–ø—Ä–µ—Ç–∏—Ä—É—é –∫–∞–∫ elapsed_time, –Ω–æ –∑–∞–ø–∏—Å—ã–≤–∞—é –≤ —Ñ–∞–π–ª —Ç–æ–ª—å–∫–æ `remain_time` (–∫–∞–∫ –æ—Å—Ç–∞–≤—à–µ–µ—Å—è, –¥–ª—è –ø—Ä–æ–¥–æ–ª–∂–µ–Ω–∏—è). –ï—Å–ª–∏ –Ω—É–∂–Ω–æ –∑–∞–ø–∏—Å—ã–≤–∞—Ç—å –∏–º–µ–Ω–Ω–æ elapsed, —Å–∫–∞–∂–∏—Ç–µ ‚Äî –¥–æ–±–∞–≤–ª—é –æ—Ç–¥–µ–ª—å–Ω—ã–π –∫–ª—é—á.
- **–î–æ–ø–æ–ª–Ω–∏—Ç–µ–ª—å–Ω–æ**:

  - –õ–æ–≥–∏ –¥–ª—è –æ—Ç–ª–∞–¥–∫–∏ (–≤–∏–¥–Ω–æ –≤ `idf.py monitor`).

  - –î–ª—è —Ç–µ—Å—Ç–∞: –ü–æ—Å–ª–µ —Ñ–ª–µ—à–∞, –∏—Å–ø–æ–ª—å–∑—É–π—Ç–µ `idf.py monitor` –∏ –∏–Ω—Å—Ç—Ä—É–º–µ–Ω—Ç—ã –≤—Ä–æ–¥–µ `esptool` –¥–ª—è —á—Ç–µ–Ω–∏—è/–∑–∞–ø–∏—Å–∏ —Ñ–∞–π–ª–∞, –∏–ª–∏ –¥–æ–±–∞–≤—å—Ç–µ –∫–æ–¥ –¥–ª—è —Å–±—Ä–æ—Å–∞.

### –ö–æ–¥ (main/main.c)
```c
#include <stdio.h>
#include <string.h>
#include "freertos/FreeRTOS.h"
#include "freertos/task.h"
#include "driver/gpio.h"
#include "esp_log.h"
#include "esp_timer.h"
#include "esp_spiffs.h"
#include "esp_vfs.h"

#define LED_PIN GPIO_NUM_8      // –ü–∏–Ω –¥–ª—è LED
#define BUTTON_PIN GPIO_NUM_1   // –ü–∏–Ω –¥–ª—è SIG –æ—Ç TTP223 (touch –∫–Ω–æ–ø–∫–∞)
#define DEBOUNCE_TIME_MS 200    // –ó–∞–¥–µ—Ä–∂–∫–∞ –¥–ª—è debounce (ms)
#define DEFAULT_MAX_TIME 60     // –î–µ—Ñ–æ–ª—Ç max_time (—Å–µ–∫), –µ—Å–ª–∏ –Ω–µ –≤ —Ñ–∞–π–ª–µ
#define CONFIG_FILE "/spiffs/config.txt"  // –ü—É—Ç—å –∫ —Ñ–∞–π–ª—É

static const char *TAG = "Touch LED Timer";
static bool led_state = false;  // –¢–µ–∫—É—â–µ–µ —Å–æ—Å—Ç–æ—è–Ω–∏–µ LED (false - –≤—ã–∫–ª)
static uint32_t max_time = DEFAULT_MAX_TIME;  // –ú–∞–∫—Å–∏–º–∞–ª—å–Ω–æ–µ –≤—Ä–µ–º—è —Å–≤–µ—á–µ–Ω–∏—è (—Å–µ–∫)
static uint32_t remain_time = DEFAULT_MAX_TIME;  // –û—Å—Ç–∞–≤—à–µ–µ—Å—è –≤—Ä–µ–º—è (—Å–µ–∫)
static uint64_t start_time = 0;  // –í—Ä–µ–º—è —Å—Ç–∞—Ä—Ç–∞ —Å–≤–µ—á–µ–Ω–∏—è (–≤ –º—Å –æ—Ç esp_timer_get_time())
static esp_timer_handle_t glow_timer;  // –•—ç–Ω–¥–ª —Ç–∞–π–º–µ—Ä–∞

// –§—É–Ω–∫—Ü–∏—è –¥–ª—è –∑–∞–ø–∏—Å–∏ –≤ —Ñ–∞–π–ª (–ø–µ—Ä–µ–∑–∞–ø–∏—Å—ã–≤–∞–µ—Ç –≤–µ—Å—å —Ñ–∞–π–ª)
static void write_config_file(void) {
    FILE *f = fopen(CONFIG_FILE, "w");
    if (f == NULL) {
        ESP_LOGE(TAG, "Failed to open file for writing");
        return;
    }
    fprintf(f, "max_time=%u\nremain_time=%u\n", max_time, remain_time);
    fclose(f);
    ESP_LOGI(TAG, "Config written: max_time=%u, remain_time=%u", max_time, remain_time);
}

// –§—É–Ω–∫—Ü–∏—è –¥–ª—è —á—Ç–µ–Ω–∏—è –∏–∑ —Ñ–∞–π–ª–∞
static void read_config_file(void) {
    FILE *f = fopen(CONFIG_FILE, "r");
    if (f == NULL) {
        ESP_LOGW(TAG, "File not found, using defaults");
        max_time = DEFAULT_MAX_TIME;
        remain_time = max_time;
        write_config_file();  // –°–æ–∑–¥–∞—ë–º —Ñ–∞–π–ª —Å –¥–µ—Ñ–æ–ª—Ç–∞–º–∏
        return;
    }

    char line[64];
    while (fgets(line, sizeof(line), f)) {
        if (sscanf(line, "max_time=%u", &max_time) == 1) {
            ESP_LOGI(TAG, "Read max_time=%u", max_time);
        } else if (sscanf(line, "remain_time=%u", &remain_time) == 1) {
            ESP_LOGI(TAG, "Read remain_time=%u", remain_time);
        }
    }
    fclose(f);

    if (remain_time > max_time) remain_time = max_time;  // –ö–æ—Ä—Ä–µ–∫—Ç–∏—Ä–æ–≤–∫–∞
}

// Callback —Ç–∞–π–º–µ—Ä–∞: –≤—ã–∫–ª—é—á–µ–Ω–∏–µ LED –ø–æ –∏—Å—Ç–µ—á–µ–Ω–∏–∏ –≤—Ä–µ–º–µ–Ω–∏
static void glow_timer_callback(void *arg) {
    led_state = false;
    gpio_set_level(LED_PIN, led_state);
    remain_time = 0;
    write_config_file();  // –ó–∞–ø–∏—Å—ã–≤–∞–µ–º remain_time=0
    ESP_LOGI(TAG, "Glow time expired. LED OFF.");
}

// –§—É–Ω–∫—Ü–∏—è –ø—Ä–µ—Ä—ã–≤–∞–Ω–∏—è (ISR) –¥–ª—è –∫–Ω–æ–ø–∫–∏
static void IRAM_ATTR button_isr_handler(void *arg) {
    static uint32_t last_press_time = 0;
    uint32_t current_time = xTaskGetTickCount() * portTICK_PERIOD_MS;

    // Debounce
    if (current_time - last_press_time > DEBOUNCE_TIME_MS) {
        if (!led_state) {  // LED –≤—ã–∫–ª—é—á–µ–Ω: –ø—ã—Ç–∞–µ–º—Å—è –≤–∫–ª—é—á–∏—Ç—å –Ω–∞ remain_time
            if (remain_time > 0) {
                led_state = true;
                gpio_set_level(LED_PIN, led_state);
                start_time = esp_timer_get_time() / 1000;  // –í—Ä–µ–º—è —Å—Ç–∞—Ä—Ç–∞ (–º—Å -> —Å–µ–∫)

                // –ó–∞–ø—É—Å–∫–∞–µ–º —Ç–∞–π–º–µ—Ä –Ω–∞ remain_time —Å–µ–∫
                esp_timer_stop(glow_timer);
                esp_timer_start_once(glow_timer, remain_time * 1000000ULL);  // –≤ us
                ESP_LOGI(TAG, "LED ON for remaining %u sec", remain_time);
            } else {
                ESP_LOGI(TAG, "No remaining time, cannot turn ON");
            }
        } else {  // LED –≤–∫–ª—é—á–µ–Ω: –≤—ã–∫–ª—é—á–∞–µ–º –∏ –æ–±–Ω–æ–≤–ª—è–µ–º remain_time
            uint64_t current_time_ms = esp_timer_get_time() / 1000;
            uint32_t elapsed = (uint32_t)(current_time_ms - start_time);
            led_state = false;
            gpio_set_level(LED_PIN, led_state);
            esp_timer_stop(glow_timer);

            if (elapsed < remain_time) {
                remain_time -= elapsed;
            } else {
                remain_time = 0;
            }
            write_config_file();  // –ó–∞–ø–∏—Å—ã–≤–∞–µ–º remain_time (–¥–ª–∏—Ç–µ–ª—å–Ω–æ—Å—Ç—å —Å–≤–µ—á–µ–Ω–∏—è —É—á—Ç–µ–Ω–∞ –∫–∞–∫ –≤—ã—á–µ—Ç)
            ESP_LOGI(TAG, "LED OFF. Elapsed: %u sec, Remaining: %u sec", elapsed, remain_time);
        }
        last_press_time = current_time;
    }
}

void app_main(void) {
    // –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è SPIFFS
    esp_vfs_spiffs_conf_t conf = {
        .base_path = "/spiffs",
        .partition_label = NULL,
        .max_files = 5,
        .format_if_mount_failed = true
    };
    esp_err_t ret = esp_vfs_spiffs_register(&conf);
    if (ret != ESP_OK) {
        ESP_LOGE(TAG, "Failed to initialize SPIFFS (%s)", esp_err_to_name(ret));
        return;
    }

    // –ß—Ç–µ–Ω–∏–µ –∫–æ–Ω—Ñ–∏–≥–∞
    read_config_file();

    // –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è GPIO –¥–ª—è LED
    gpio_config_t io_conf_led = {
        .intr_type = GPIO_INTR_DISABLE,
        .mode = GPIO_MODE_OUTPUT,
        .pin_bit_mask = (1ULL << LED_PIN),
        .pull_down_en = 0,
        .pull_up_en = 0
    };
    gpio_config(&io_conf_led);
    gpio_set_level(LED_PIN, 0);

    // –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è GPIO –¥–ª—è –∫–Ω–æ–ø–∫–∏
    gpio_config_t io_conf_button = {
        .intr_type = GPIO_INTR_POSEDGE,
        .mode = GPIO_MODE_INPUT,
        .pin_bit_mask = (1ULL << BUTTON_PIN),
        .pull_down_en = 1,
        .pull_up_en = 0
    };
    gpio_config(&io_conf_button);

    // –†–µ–≥–∏—Å—Ç—Ä–∞—Ü–∏—è –ø—Ä–µ—Ä—ã–≤–∞–Ω–∏—è
    gpio_install_isr_service(0);
    gpio_isr_handler_add(BUTTON_PIN, button_isr_handler, NULL);

    // –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è —Ç–∞–π–º–µ—Ä–∞
    const esp_timer_create_args_t timer_args = {
        .callback = &glow_timer_callback,
        .name = "glow_timer"
    };
    esp_timer_create(&timer_args, &glow_timer);

    ESP_LOGI(TAG, "System initialized. Max time: %u sec, Remaining: %u sec. Touch to toggle.", max_time, remain_time);

    // –ë–µ—Å–∫–æ–Ω–µ—á–Ω—ã–π —Ü–∏–∫–ª
    while (1) {
        vTaskDelay(pdMS_TO_TICKS(1000));
    }
}
```

### –¢–µ—Å—Ç–∏—Ä–æ–≤–∞–Ω–∏–µ –∏ –æ—Ç–ª–∞–¥–∫–∞:
